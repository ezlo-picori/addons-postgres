#include <tunables/global>

profile example flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # PostgreSQL
  /{,var/}run/postgresql/** rw,

  # Access to options.json and other files within your addon
  /data/** rw,

  # PostgreSQL binaries
  /usr/bin/initdb cx -> initdb,
  /usr/bin/psql cx -> psql,

  profile initdb=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    /data/databases/** rw,
  }

  profile psql=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    #include <abstractions/nameservice>
    #include <abstractions/ssl_keys>

    /etc/postgresql/** r,
    /usr/share/postgresql/** r,
    /var/lib/postgresql/** rwl,
    /{,var/}run/postgresql/** rw,

    owner @{PROC}/[0-9]*/oom_adj rw,
  }
}
